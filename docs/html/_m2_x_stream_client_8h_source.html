1102 static inline int m2x_fill_iso8601_timestamp(int32_t timestamp, int32_t milli,
1103  char* buffer, int* length) {
1104  int16_t year;
1105  int8_t month, month_length;
1106  int32_t day;
1107  int8_t hour, minute, second;
1108 
1109  if (*length < SIZE_ISO_8601) {
1110  *length = SIZE_ISO_8601;
1111  return E_BUFFER_TOO_SMALL;
1112  }
1113 
1114  second = timestamp % 60;
1115  timestamp /= 60; // now it is minutes
1116 
1117  minute = timestamp % 60;
1118  timestamp /= 60; // now it is hours
1119 
1120  hour = timestamp % 24;
1121  timestamp /= 24; // now it is days
1122 
1123  year = 0;
1124  day = 0;
1125  while ((day += days_in_year(year)) <= timestamp) {
1126  year++;
1127  }
1128  day -= days_in_year(year);
1129  timestamp -= day; // now it is days in this year, starting at 0
1130 
1131  day = 0;
1132  month_length = 0;
1133  for (month = 0; month < 12; month++) {
1134  if (month == 1) {
1135  // February
1136  month_length = is_leap_year(year) ? 29 : 28;
1137  } else {
1138  month_length = MONTH_DAYS[month];
1139  }
1140 
1141  if (timestamp >= month_length) {
1142  timestamp -= month_length;
1143  } else {
1144  break;
1145  }
1146  }
1147  year = 1970 + year;
1148  month++; // offset by 1
1149  day = timestamp + 1;
1150

1152 
1153  // NOTE: It seems the snprintf implementation in Arduino has bugs,
1154  // we have to manually piece the string together here.
1155 #define INT_TO_STR(v_, width_) \
1156  for (j = 0; j < (width_); j++) { \
1157  buffer[i + (width_) - 1 - j] = '0' + ((v_) % 10); \
1158  (v_) /= 10; \
1159  } \
1160  i += (width_)

1163  buffer[i++] = '-';
1164  INT_TO_STR(month, 2);
1165  buffer[i++] = '-';
1166  INT_TO_STR(day, 2);
1167  buffer[i++] = 'T';
1168  INT_TO_STR(hour, 2);
1169  buffer[i++] = ':';
1170  INT_TO_STR(minute, 2);
1171  buffer[i++] = ':';
1172  INT_TO_STR(second, 2);
1173  buffer[i++] = '.';
1174  INT_TO_STR(milli, 3);
1175  buffer[i++] = 'Z';
1176  buffer[i++] = '\0';
1177
